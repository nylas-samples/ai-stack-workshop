  const text = `

  `
    
  const embedding = await createEmebedding(
    text, 
    { pooling: 'mean', normalize: true }
  )
  
  embeddings.push({id: message.id, data: embedding.data })
}

Part 2/3: Search Embeddings

import { calculateSimilarity } from './utils/calculateSimilarity.js'

async function semanticSearch(query) {
  const queryEmbedding = await createEmebedding(
    query, 
    { pooling: 'mean', normalize: true }
  )

  const results = []
  
  embeddings.map(embedding => {
    const similarity = calculateSimilarity(
      queryEmbedding.data,
      embedding.data
    )
    
    results.push({ id: embedding.id, similarity })
  })
  
  results.sort((a, b) => b.similarity - a.similarity)
  
  return results.filter(result => result.similarity >= 0.30)
}

Part 3/3: User Input Loop

import readline from 'readline'

async function promptSearch() {
  rl.question("What's your query? ", async (query) => {
    if (query.toLowerCase() === 'exit') {
      rl.close()
      return
    }

    const searchResults = await semanticSearch(query)

    for (const result of searchResults.slice(0, 200)) {
        const message = messages.data.find(e => e.id === result.id)
        console.log("Message Subject:", message.subject)
        console.log("Similarity score:", result.similarity)
    }

    promptSearch()
  });
}
